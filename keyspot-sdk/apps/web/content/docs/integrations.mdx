# Framework Integrations

## LangChain

```typescript
import { withKeySpot } from '@roadsidelab/keyspot-frameworks';

const guardedChain = withKeySpot(myChain, guard);
```

## Anthropic

```typescript
import { wrapAnthropic } from '@roadsidelab/keyspot-adapters/anthropic';

const client = wrapAnthropic(anthropic, guard);
```

## OpenAI

```typescript
import { wrapOpenAI } from '@roadsidelab/keyspot-adapters/openai';

const client = wrapOpenAI(openai, guard);
```

## Express Middleware

```typescript
import { keyspotMiddleware } from '@roadsidelab/keyspot-server';

app.use(keyspotMiddleware(guard));
```
